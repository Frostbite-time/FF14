	NAND = {
		has_council_position = councillor_court_chaplain
		faith = { has_doctrine_parameter = clerical_appointment_fixed }
	}
	NOT = {
		OR = { 
			has_council_position = councillor_spouse
			AND = {
				this = liege.diarch
				liege = { has_diarchy_active_parameter = diarchy_is_vizierate }
			}
		}
	}
	OR = {
		is_ruler = yes
		AND = {
			is_female = yes
			$内阁所有者$.faith = { NOT = { has_doctrine_parameter = male_dominated_council } }
		}
		AND = {
			is_male = yes
			$内阁所有者$.faith = { NOT = { has_doctrine_parameter = female_dominated_council } }
		}
	}
}
